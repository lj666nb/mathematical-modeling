## 二、新增数据模型

### 2.1 MathModel — 数学模型知识库表 🆕

```python
class MathModel(Base):
    """数学模型知识库——替代 Experiment 表的数学建模版本"""
    __tablename__ = "math_model"

    id            = Column(Integer, PK, autoincrement=True)
    category      = Column(String(50), nullable=False,
                     comment="分类: optimization/prediction/evaluation/clustering/differential/statistics/graph/stochastic")
    name          = Column(String(200), nullable=False, comment="模型名称")
    summary       = Column(String(500), comment="一句话概述")
    definition    = Column(Text, comment="模型定义（Markdown）")
    formula       = Column(Text, comment="核心公式（LaTeX）")
    applicable    = Column(Text, comment="适用场景（JSON数组）")
    typical_methods = Column(Text, comment="典型方法（JSON数组）")
    example_problem = Column(Text, comment="例题描述")
    example_solution = Column(Text, comment="例题解答")
    code_template = Column(Text, comment="Python 代码模板")
    difficulty    = Column(Integer, default=1, comment="难度 1-5")
    tags          = Column(String(500), comment="标签（逗号分隔）")
    related_ids   = Column(String(200), comment="关联模型ID（逗号分隔）")
    created_at    = Column(DateTime, default=func.now())
    updated_at    = Column(DateTime, default=func.now(), onupdate=func.now())
```

### 2.2 CompetitionProject — 竞赛项目表 🆕

```python
class CompetitionProject(Base):
    """竞赛训练——项目管理"""
    __tablename__ = "competition_project"

    id            = Column(String(36), PK, default=uuid4_str,
                     comment="项目ID (UUID)")
    user_id       = Column(Integer, FK("user.id", ondelete="CASCADE"))
    title         = Column(String(200), nullable=False, comment="项目名称")
    competition   = Column(String(50), comment="竞赛: CUMCM/MCM/ICM/Other")
    description   = Column(Text, comment="项目描述")
    status        = Column(String(20), default="created",
                     comment="created/initialized/running/paused/completed")
    current_stage = Column(String(5), nullable=True,
                     comment="当前阶段: S0-S8")
    paper_output_dir = Column(String(500), comment="paper_output/ 路径")
    created_at    = Column(DateTime, default=func.now())
    updated_at    = Column(DateTime, default=func.now(), onupdate=func.now())

    # 关联
    stages        = relationship("CompetitionStage", back_populates="project",
                                 cascade="all, delete-orphan",
                                 order_by="CompetitionStage.stage_order")
```

### 2.3 CompetitionStage — 竞赛阶段记录表 🆕

```python
class CompetitionStage(Base):
    """竞赛训练——每阶段执行记录"""
    __tablename__ = "competition_stage"

    id            = Column(Integer, PK, autoincrement=True)
    project_id    = Column(String(36), FK("competition_project.id", ondelete="CASCADE"))
    stage         = Column(String(5), nullable=False, comment="阶段: S0-S8")
    stage_order   = Column(Integer, comment="排序: 0-8")
    status        = Column(String(20), default="pending",
                     comment="pending/running/completed/failed/skipped")
    stdout        = Column(Text, comment="脚本标准输出")
    stderr        = Column(Text, comment="脚本标准错误")
    contract_json = Column(Text, comment="契约JSON（阶段产物摘要）")
    warnings      = Column(Text, comment="警告信息（JSON数组）")
    elapsed_ms    = Column(Integer, comment="执行耗时(毫秒)")
    started_at    = Column(DateTime, comment="开始时间")
    completed_at  = Column(DateTime, comment="完成时间")

    # 关联
    project = relationship("CompetitionProject", back_populates="stages")
```

class MathCase(Base):
