### 2.2 chatStore — 对话状态 🆕

```js
// stores/chat.js
export const useChatStore = defineStore('chat', () => {
  // ===== 状态 =====
  const currentAgent = ref('math-qa')       // 当前选中的Agent
  const sessionId = ref(null)               // 当前会话ID
  const messages = ref([])                  // 消息列表
  const isStreaming = ref(false)            // 是否正在流式接收
  const streamingContent = ref('')          // 流式接收中的内容

  // ===== 可用Agent列表 =====
  const agents = ref([
    {
      type: 'math-qa',
      icon: '🧮',
      name: '数学答疑',
      desc: '数学概念解释、公式推导、方法对比',
      conversationCount: '42.5k'
    },
    {
      type: 'modeling-guide',
      icon: '📐',
      name: '建模引导',
      desc: '问题类型识别、模型推荐、建模思路引导',
      conversationCount: '38.2k'
    },
    {
      type: 'code-assist',
      icon: '💻',
      name: '代码辅助',
      desc: 'Python代码纠错、库推荐、脚手架生成',
      conversationCount: '29.8k'
    },
    {
      type: 'paper-review',
      icon: '📝',
      name: '论文评审',
      desc: '论文结构审查、评分标准对齐、改进建议',
      conversationCount: '—',
      available: false  // P1
    }
  ])

  // ===== 计算属性 =====
  const currentAgentInfo = computed(() =>
    agents.value.find(a => a.type === currentAgent.value)
  )

  // ===== 方法 =====
  function selectAgent(type) { currentAgent.value = type }
  function newSession() {
    sessionId.value = null
    messages.value = []
    streamingContent.value = ''
  }

  async function sendMessage(content, codeContext = null) {
    // 1. 添加用户消息到列表
    messages.value.push({ role: 'user', content, time: Date.now() })

    // 2. 调用API (WebSocket流式)
    isStreaming.value = true
    streamingContent.value = ''

    const ws = new WebSocket(
      `/api/chat/ws?agent_type=${currentAgent.value}&session_id=${sessionId.value || ''}`
    )

    ws.onmessage = (e) => {
      const data = JSON.parse(e.data)
      if (data.type === 'token') {
        streamingContent.value += data.content
      } else if (data.type === 'done') {
        messages.value.push({
          role: 'assistant',
          content: streamingContent.value,
          time: Date.now(),
          suggestions: data.suggestions || [],
          chatId: data.chat_id
        })
        sessionId.value = data.session_id
        isStreaming.value = false
        streamingContent.value = ''
        ws.close()
      }
    }

    ws.onerror = () => {
      isStreaming.value = false
      messages.value.push({
        role: 'assistant',
        content: '⚠️ 连接失败，请重试',
        time: Date.now(),
        error: true
      })
    }
  }

  function stopStreaming() {
    // 关闭WebSocket连接
    isStreaming.value = false
  }

  async function loadHistory(sessionId) {
    const res = await chatApi.getHistory({ session_id: sessionId })
    messages.value = res.history.map(h => ({
      role: 'user', content: h.user_message, time: h.created_at
    })).concat(res.history.map(h => ({
      role: 'assistant', content: h.agent_message, time: h.created_at
    })))
  }

  return {
    currentAgent, sessionId, messages, isStreaming, streamingContent,
    agents, currentAgentInfo,
    selectAgent, newSession, sendMessage, stopStreaming, loadHistory
  }
})
```

### 2.3 competitionStore — 竞赛状态 🆕

```js
// stores/competition.js
export const useCompetitionStore = defineStore('competition', () => {
  // ===== 状态 =====
  const projects = ref([])                   // 项目列表
  const currentProject = ref(null)           // 当前查看的项目
  const stages = ref({})                     // 阶段状态 {S0: {...}, S1: {...}, ...}
  const isStageRunning = ref(false)          // 是否有阶段正在执行
  const runningStage = ref(null)             // 正在执行的阶段
  const stageStreamOutput = ref('')          // 阶段执行流式输出

  // ===== 计算属性 =====
  const currentStage = computed(() =>
    currentProject.value?.current_stage || null
  )
  const stagesList = computed(() =>
    ['S0','S1','S2','S3','S4','S5','S6','S7','S8'].map(s => ({
      stage: s,
      ...stages.value[s],
      isCurrent: s === currentProject.value?.current_stage
    }))
  )
  const evidenceGateStatus = computed(() => stages.value['S6']?.status)
  const formatGateStatus = computed(() => stages.value['S7']?.status)
  const canExport = computed(() =>
    stages.value['S7']?.status === 'completed' && formatGateStatus.value === 'completed'
  )

  // ===== 方法 =====
  async function fetchProjects() {
    const res = await competitionApi.getProjects()
    projects.value = res.projects
  }

  async function createProject(data) {
    const res = await competitionApi.createProject(data)
    projects.value.unshift(res)
    return res
  }

  async function fetchProjectDetail(projectId) {
    currentProject.value = await competitionApi.getProject(projectId)
    const stageRes = await competitionApi.getStages(projectId)
    stages.value = stageRes.stages
  }

  async function runStage(projectId, stage, params = {}) {
    isStageRunning.value = true
    runningStage.value = stage
    stages.value[stage] = { ...stages.value[stage], status: 'running' }

    try {
      const res = await competitionApi.runStage(projectId, stage, params)
      stages.value[stage] = {
        status: res.status,
        output: res.output,
        contract: res.contract,
        elapsed: res.elapsed_seconds
      }
      if (res.next_stage) {
        currentProject.value.current_stage = res.next_stage
      }
      return res
    } catch (e) {
      stages.value[stage] = { ...stages.value[stage], status: 'failed' }
      throw e
    } finally {
      isStageRunning.value = false
      runningStage.value = null
    }
  }

  async function runAuto(projectId, untilStage = 'S7') {
    const res = await competitionApi.runAuto(projectId, { until_stage: untilStage })
    // 轮询状态或WebSocket接收进度
    return res.task_id
  }

  async function downloadPaper(projectId) {
    const blob = await competitionApi.exportPaper(projectId)
    const url = URL.createObjectURL(blob)
    const a = document.createElement('a')
    a.href = url
    a.download = `final_paper_${projectId}.docx`
    a.click()
    URL.revokeObjectURL(url)
  }

  return {
    projects, currentProject, stages, isStageRunning, runningStage,
    stageStreamOutput,
    currentStage, stagesList, evidenceGateStatus, formatGateStatus, canExport,
    fetchProjects, createProject, fetchProjectDetail, runStage, runAuto, downloadPaper
  }
})
```

### 2.4 workspaceStore — 工作台状态 🆕

```js
// stores/workspace.js
export const useWorkspaceStore = defineStore('workspace', () => {
  const code = ref('')                       // 当前编辑的代码
  const language = ref('python')
  const isExecuting = ref(false)            // 是否正在执行
  const lastResult = ref(null)              // 最近执行结果
  const files = ref([])                     // 工作区文件列表
  const charts = ref([])                    // 执行生成的图表

  async function executeCode(codeStr) {
    isExecuting.value = true
    lastResult.value = null
    charts.value = []

    try {
      const res = await workspaceApi.execute({
        code: codeStr || code.value,
        language: language.value,
        timeout: 30
      })
      lastResult.value = {
        stdout: res.stdout,
        stderr: res.stderr,
        returncode: res.returncode,
        execution_time_ms: res.execution_time_ms
      }
      charts.value = res.charts || []
      return lastResult.value
    } finally {
      isExecuting.value = false
    }
  }

  async function uploadFiles(fileList) {
    const formData = new FormData()
    fileList.forEach(f => formData.append('files', f))
    await workspaceApi.upload(formData)
    await refreshFiles()
  }

  async function refreshFiles() {
    const res = await workspaceApi.getFiles()
    files.value = res.files
  }

  return {
    code, language, isExecuting, lastResult, files, charts,
    executeCode, uploadFiles, refreshFiles
  }
})
```

## 三、持久化策略

| Store | 持久化内容 | 存储位置 | 理由 |
|------|----------|------|------|
| `userStore` | token, user | localStorage | 跨会话保持登录态 |
| `chatStore` | 最近会话ID | localStorage | 刷新后恢复当前会话 |
| `chatStore` | 对话历史 | 服务端DB（通过API） | 多设备同步、不丢失 |
| `competitionStore` | 无 | — | 数据通过API实时获取 |
| `workspaceStore` | 代码草稿 | sessionStorage | 页面刷新可恢复，关闭后清除 |

### 持久化实现（pinia-plugin-persistedstate）

```js
// 仅 userStore 需要持久化
// stores/user.js 保持不变（手动 localStorage）
// chatStore 的会话ID可加轻量持久化:

// 在 chatStore 中:
watch(sessionId, (val) => {
  if (val) sessionStorage.setItem('last_session_id', val)
})
```
